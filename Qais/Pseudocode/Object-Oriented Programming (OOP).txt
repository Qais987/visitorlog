TASK 1 Movie Ticket Booking System  

CLASS Movie
    title : TEXT
    ticketPrice : NUMBER
    totalSeats : NUMBER
    bookedSeats : NUMBER

    METHOD GET_AVAILABLE_SEATS()
        RETURN totalSeats - bookedSeats
    END METHOD

    METHOD BOOK_TICKETS(amount)
        IF amount LESS THAN OR EQUAL TO GET_AVAILABLE_SEATS() THEN
            bookedSeats TO bookedSeats + amount
            DISPLAY "Booking successful"
            DISPLAY "Remaining seats: " + GET_AVAILABLE_SEATS()
        ELSE
            DISPLAY "Not enough seats available"
        ENDIF
    END METHOD

    METHOD DISPLAY_INFO()
        DISPLAY "Movie: " + title
        DISPLAY "Ticket price: " + ticketPrice
        DISPLAY "Available seats: " + GET_AVAILABLE_SEATS()
    END METHOD
END CLASS


CLASS RegularMovie EXTENDS Movie
END CLASS


CLASS PremiumMovie EXTENDS Movie

    METHOD GET_TICKET_PRICE()
        RETURN ticketPrice + (ticketPrice * 0.20)
    END METHOD

END CLASS


CREATE m1 AS RegularMovie
m1.title TO "Avatar"
m1.ticketPrice TO 10
m1.totalSeats TO 100
m1.bookedSeats TO 0

CREATE m2 AS PremiumMovie
m2.title TO "Avengers"
m2.ticketPrice TO 10
m2.totalSeats TO 50
m2.bookedSeats TO 0


CALL m1.BOOK_TICKETS(5)
CALL m2.BOOK_TICKETS(10)

CALL m1.DISPLAY_INFO()
CALL m2.DISPLAY_INFO()

TASK 2 Employee Payroll System  


CLASS Employee
    name : TEXT
    baseSalary : NUMBER
    hoursWorked : NUMBER
    deductions : NUMBER

    METHOD ADD_HOURS(hours)
        IF hours GREATER THAN 0 THEN
            hoursWorked ← hoursWorked + hours
        ELSE
            DISPLAY "Invalid hours"
        ENDIF
    END METHOD

    METHOD ADD_DEDUCTION(amount)
        IF amount GREATER THAN 0 THEN
            deductions TO deductions + amount
        ELSE
            DISPLAY "Invalid deduction"
        ENDIF
    END METHOD

    METHOD CALCULATE_SALARY()
        RETURN baseSalary
    END METHOD

    METHOD DISPLAY_INFO()
        DISPLAY "Employee: " + name
        DISPLAY "Hours: " + hoursWorked
        DISPLAY "Salary: " + CALCULATE_SALARY()
    END METHOD
END CLASS


CLASS FullTimeEmployee EXTENDS Employee

    METHOD CALCULATE_SALARY()
        IF hoursWorked GREATER THAN 40 THEN
            overtime TO hoursWorked - 40
            overtimePay TO overtime * (baseSalary / 40) * 1.5
            salary TO baseSalary + overtimePay - deductions
        ELSE
            salary TO baseSalary - deductions
        ENDIF
        RETURN salary
    END METHOD

END CLASS


CLASS PartTimeEmployee EXTENDS Employee

    METHOD CALCULATE_SALARY()
        hourlyRate TO baseSalary / 40
        salary TO hoursWorked * hourlyRate - deductions
        RETURN salary
    END METHOD

END CLASS


CREATE emp1 AS FullTimeEmployee
emp1.name TO "Ahmed"
emp1.baseSalary TO 2000
emp1.hoursWorked TO 0
emp1.deductions TO 0

CALL emp1.ADD_HOURS(45)
CALL emp1.ADD_DEDUCTION(100)
CALL emp1.DISPLAY_INFO()

TASK 3 Online Shopping System  

CLASS Product
    name : TEXT
    price : NUMBER
    stock : NUMBER

    METHOD BUY_PRODUCT(quantity)
        IF quantity LESS THAN OR EQUAL TO stock THEN
            stock TO stock - quantity
            DISPLAY "Purchase successful"
            DISPLAY "Remaining stock: " + stock
        ELSE
            DISPLAY "Not enough stock"
        ENDIF
    END METHOD

    METHOD GET_FINAL_PRICE()
        RETURN price
    END METHOD

    METHOD DISPLAY_INFO()
        DISPLAY name
        DISPLAY price
        DISPLAY stock
    END METHOD
END CLASS


CLASS DigitalProduct EXTENDS Product

    METHOD GET_FINAL_PRICE()
        RETURN price
    END METHOD

END CLASS


CLASS PhysicalProduct EXTENDS Product
    shippingCost : NUMBER

    METHOD GET_FINAL_PRICE()
        RETURN price + shippingCost
    END METHOD
END CLASS


CREATE p1 AS PhysicalProduct
p1.name TO "Laptop"
p1.price TO 1000
p1.stock TO 10
p1.shippingCost TO 50

CALL p1.BUY_PRODUCT(2)

TASK 4 Hospital Patient Management System  


CLASS Patient
    name : TEXT
    age : NUMBER
    symptoms : LIST
    medications : LIST
    isAdmitted : BOOLEAN

    METHOD ADD_SYMPTOM(symptom)
        ADD symptom TO symptoms
        DISPLAY "Symptom added"
    END METHOD

    METHOD ADMIT_PATIENT()
        isAdmitted TO TRUE
        DISPLAY "Patient admitted"
    END METHOD

    METHOD PRESCRIBE_MEDICINE(medicine)
        IF isAdmitted EQUAL TO TRUE THEN
            ADD medicine TO medications
            DISPLAY "Medicine prescribed"
        ELSE
            DISPLAY "Patient must be admitted first"
        ENDIF
    END METHOD

    METHOD GET_PRIORITY()
        IF SIZE(symptoms) GRATER THAN OR EQUAL TO 4 THEN
            RETURN "Critical"
        ELSE IF SIZE(symptoms) GRATER THAN OR EQUAL TO 2 THEN
            RETURN "Medium"
        ELSE
            RETURN "Low"
        ENDIF
    END METHOD
END CLASS

TASK 5 Smart Parking Management System  


CLASS Vehicle
    plateNumber : TEXT
    entryTime : NUMBER
    exitTime : NUMBER
    hourlyRate : NUMBER

    METHOD SET_EXIT_TIME(time)
        exitTime TO time
    END METHOD

    METHOD CALCULATE_PARKING_TIME()
        RETURN exitTime - entryTime
    END METHOD

    METHOD CALCULATE_FEE()
        RETURN 0
    END METHOD
END CLASS


CLASS Car EXTENDS Vehicle

    METHOD CALCULATE_FEE()
        RETURN CALCULATE_PARKING_TIME() * hourlyRate
    END METHOD

END CLASS


CLASS Truck EXTENDS Vehicle

    METHOD CALCULATE_FEE()
        RETURN CALCULATE_PARKING_TIME() * hourlyRate * 1.5
    END METHOD

END CLASS


CREATE car1 AS Car
car1.plateNumber TO "123A"
car1.entryTime TO 2
car1.hourlyRate TO 5

CALL car1.SET_EXIT_TIME(5)

DISPLAY car1.CALCULATE_FEE()

TASK 7 Library Borrowing and Membership System  

CLASS Book
    title : TEXT
    author : TEXT
    isBorrowed : BOOLEAN

    METHOD BORROW_BOOK(member)
        IF isBorrowed EQUAL TO FALSE THEN
            isBorrowed TO TRUE
            DISPLAY "Book borrowed by " + member
        ELSE
            DISPLAY "Book already borrowed"
        ENDIF
    END METHOD

    METHOD RETURN_BOOK()
        isBorrowed TO FALSE
        DISPLAY "Book returned"
    END METHOD
END CLASS


CLASS Member
    name : TEXT
    borrowedBooks : LIST
    borrowLimit : NUMBER

    METHOD BORROW(book)
        IF SIZE(borrowedBooks) < borrowLimit THEN
            CALL book.BORROW_BOOK(name)
            ADD book TO borrowedBooks
        ELSE
            DISPLAY "Borrow limit reached"
        ENDIF
    END METHOD
END CLASS


CLASS RegularMember EXTENDS Member
END CLASS


CLASS PremiumMember EXTENDS Member
END CLASS


CREATE book1 AS Book
book1.title TO "Python Basics"
book1.author TO "John"

CREATE mem1 AS RegularMember
mem1.name TO "Ali"
mem1.borrowLimit TO 3

CALL mem1.BORROW(book1)
